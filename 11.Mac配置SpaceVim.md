# Mac配置SpaceVim

- 安装

  ```bash
  SpaceVim 安装脚本 : V 1.7.0-dev
  
    这是 SpaceVim 初始化脚本，可用于定制安装、更新及卸载 SpaceVim。
  
  使用
  
    curl -sLf https://spacevim.org/cn/install.sh | bash -s -- [选项] [对象]
  
  所有选项
  
   -i, --install            为 vim 和 neovim 安装 SpaceVim
   -v, --version            显示当前安装脚本的版本
   -u, --uninstall          卸载 SpaceVim
   -c, --checkRequirements  检测环境依赖
  
  使用示例
  
      默认同时为 vim 和 neovim 安装 SpaceVim
  
          curl -sLf https://spacevim.org/cn/install.sh | bash
  
      只为 vim 或者 neovim 单独安装 SpaceVim
  
          curl -sLf https://spacevim.org/cn/install.sh | bash -s -- --install vim
          curl -sLf https://spacevim.org/cn/install.sh | bash -s -- --install neovim
  
      卸载 SpaceVim
  
          curl -sLf https://spacevim.org/cn/install.sh | bash -s -- --uninstall
  ```

- 安装vimproc 

  ```bash
  :VimProcInstall
  ```

- 配置

  ```toml
  #=============================================================================
  # dark_powered.toml --- dark powered configuration example for SpaceVim
  # Copyright (c) 2016-2020 Wang Shidong & Contributors
  # Author: Wang Shidong < wsdjeg at 163.com >
  # URL: https://spacevim.org
  # License: GPLv3
  #=============================================================================
  
  # All SpaceVim option below [option] section
  [options]
      # 默认主题
      colorscheme = "gruvbox"
      colorscheme_bg = "dark"
      # 使用真色
      enable_guicolors = true
      # 状态栏分割符
      statusline_separator = "arrow"
      # 设置非活动窗口状态栏上的分割符号形状
      statusline_iseparator = "bar"
      buffer_index_type = 4
      # 标签栏上的文件类型图标
      enable_tabline_filetype_icon = true
      # 是否在状态栏上显示当前模式，默认情况下，不显示 Normal/Insert 等
      enable_statusline_mode = true
      # 字体使用
      guifont = "SourceCodePro Nerd Font Mono:h11"
      # 设置默认锁进
      default_indent = 4
      expand_tab = true
      # 自动补全方法
  
  # Enable autocomplete layer
  [[layers]]
      name = 'autocomplete'
      auto_completion_return_key_behavior = "complete"
      auto_completion_tab_key_behavior = "cycle"
  
  [[layers]]
      name = 'shell'
      default_position = 'top'
      default_height = 30
  
  [[layers]]
      name = "incsearch"
  
  [[layers]]
      name = "colorscheme"
      random_theme = true
      frequency = "daily"
  
  [[layers]]
      name = "lang#c"
      clang_executable = "/usr/bin/clang"
      [layer.clang_std]
          c = "c11"
          cpp = "c++1z"
          objc = "c11"
          objcpp = "c++1z"
  
  [[layers]]
      name = "lang#go"
  
  [[layers]]
      name = "lang#html"
  
  [[layers]]
    name = "lang#javascript"
  
  [[layers]]
      name = 'lang#java'
      java_file_head = [
          '/**',
          ' * @author : `fnamemodify(expand("~"), ":t")`',
          ' * @created : `strftime("%Y-%m-%d")`',
          '**/',
          ''
      ]
  
  [[layers]]
      name = "lang#python"
      python_file_head = [
          '#!/usr/bin/env python',
          '# -*- coding: utf-8 -*-',
          '',
          ''
      ]
  
  [[layers]]
      name = "lang#markdown"
  
  [[layers]]
      name = "lang#vim"
  
  [[layers]]
      name = "tags"
  
  [[layers]]
    name = "gtags"
    gtagslabel = "ctags"
  ```
